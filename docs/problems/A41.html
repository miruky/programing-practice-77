# =====================================================================
# A41: 3 連続パターンの検出
# ---------------------------------------------------------------------
# 【概要】文字列 S 内に "RRR" または "BBB" が部分文字列として現れるか。
# 【アルゴリズム】長さ 3 の窓で線形に走査する。
# 【計算量】O(N)
# =====================================================================

# --- 入力 -----------------------------------------------------------
N = int(input())
S = input()

# --- 走査 -----------------------------------------------------------
Answer = False
for i in range(0, N - 2):
    if S[i] == 'R' and S[i + 1] == 'R' and S[i + 2] == 'R':
        Answer = True
    if S[i] == 'B' and S[i + 1] == 'B' and S[i + 2] == 'B':
        Answer = True

# --- 出力 -----------------------------------------------------------
if Answer == True:
    print("Yes")
else:
    print("No")
