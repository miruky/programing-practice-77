# =====================================================================
# C17: Strange Data Structure (力試し問題)
# ---------------------------------------------------------------------
# 【概要】特殊なキュークエリ。
#         A x: 最後尾に x を追加
#         B x: 中央 (前半デックの末尾) に x を追加
#         C : 先頭を取り除く
#         D : 先頭を出力する
# 【アルゴリズム】2 つのデック Z1, Z2 を使う:
#         Z1 = 前半、Z2 = 後半。サイズ差が ±1 以内になるよう毎クエリ後に
#         リバランスする。先頭 = Z1.front()。
# 【計算量】各クエリ償却 O(1)。
# =====================================================================
from collections import deque
import sys
input = sys.stdin.readline                # 高速化

# --- 入力 -----------------------------------------------------------
Q = int(input())
queries = [input().split() for _ in range(Q)]

# --- 2 つのデック ---------------------------------------------------
Z1 = deque()                              # 前半 (先頭側)
Z2 = deque()                              # 後半

out = []
for q in queries:
    kind = q[0]
    if kind == 'A':
        Z2.append(q[1])
    elif kind == 'B':
        Z1.append(q[1])
    elif kind == 'C':
        Z1.popleft()
    elif kind == 'D':
        out.append(Z1[0])

    # --- リバランス: Z1 が Z2 より多すぎる場合 ----------------------
    while len(Z1) - len(Z2) >= 2:
        Z2.appendleft(Z1.pop())
    # --- リバランス: Z2 が Z1 より多すぎる場合 ----------------------
    while len(Z1) - len(Z2) <= -1:
        Z1.append(Z2.popleft())

print("\n".join(out))
