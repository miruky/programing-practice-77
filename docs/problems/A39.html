A39.py ─ Python 3 
# =====================================================================
# A39: 区間スケジューリング
# ---------------------------------------------------------------------
# 【概要】N 個の区間 [l, r] から、できるだけ多くの非重複区間を選ぶ。
# 【アルゴリズム】終了時刻でソートして貪欲に選択する古典問題。
#   "終了時刻が早い区間ほど後の選択肢を残せる" という直観が根拠。
# 【計算量】O(N log N)
# =====================================================================

# --- 入力 (タプル順を (r, l) にして終了時刻で昇順ソートしやすくする) -----
N = int(input())
A = []
for i in range(N):
    l, r = map(int, input().split())
    A.append([r, l])

# 終了時刻 r の昇順にソート (l は 2 番目の要素なので副次的)
A.sort()

# --- 貪欲: 直前の終了時刻以降に始まる区間を順に採用 ----------------
CurrentTime = 0
Answer = 0
for i in range(N):
    if CurrentTime <= A[i][1]:  # i 番目の区間の開始時刻 >= 現在時刻
        CurrentTime = A[i][0]   # 区間の終了時刻に更新
        Answer += 1

# --- 出力 -----------------------------------------------------------
print(Answer)
