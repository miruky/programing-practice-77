# =====================================================================
# A74: 転倒数に分解 (スライドパズル)
# ---------------------------------------------------------------------
# 【概要】N×N の盤面上のスライドパズルで、目標配置にする最小操作回数を求める。
# 【アルゴリズム】考察 (分解)
#   - 操作は行と列で独立: 行を並べ替える操作と列を並べ替える操作。
#   - 各操作は隣接スワップなので、各方向の最小操作数は "転倒数"。
#   - 行と列に分け、転倒数を独立に計算して合計するのが答え。
# 【計算量】O(N^2)
# =====================================================================

# --- 入力 -----------------------------------------------------------
N = int(input())
P = [list(map(int, input().split())) for _ in range(N)]

# --- 行・列に分解 -------------------------------------------------
# 0 でない値を持つ行・列が一意に決まる (一直線に並んでいる構造)
X = [None] * N
Y = [None] * N
for i in range(N):
    for j in range(N):
        if P[i][j] != 0:
            X[i] = P[i][j]  # 行 i の代表値
            Y[j] = P[i][j]  # 列 j の代表値


def inversion(A):
    """配列 A の転倒数 (i<j で A[i]>A[j] となる組数)"""
    answer = 0
    for i in range(len(A)):
        for j in range(i + 1, len(A)):
            if A[i] > A[j]:
                answer += 1
    return answer


# --- 行転倒数 + 列転倒数 = 必要操作回数 ----------------------------
print(inversion(X) + inversion(Y))

N = int(input())
