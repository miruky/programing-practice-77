# =====================================================================
# B03: Supermarket 1 (A03 応用)
# ---------------------------------------------------------------------
# 【概要】N 個の値段 A[0..N-1] の中から異なる 3 つを選んで合計を 1000 に
#         できるかを判定する。
# 【アルゴリズム】三重ループで i<j<k を満たす全組合せを試す。
# 【計算量】O(N³)。N ≦ 100 なら 10^6 で十分高速。
# 【ポイント】「異なる」3 つにするには i<j<k と添字を厳密に大小制約する。
# =====================================================================

# --- 入力 -----------------------------------------------------------
N = int(input())
A = list(map(int, input().split()))

# --- 三重ループ全探索 -----------------------------------------------
Answer = False
for i in range(N):
    for j in range(i + 1, N):          # j は i より大きい添字のみ
        for k in range(j + 1, N):      # k は j より大きい添字のみ
            if A[i] + A[j] + A[k] == 1000:
                Answer = True

# --- 出力 -----------------------------------------------------------
print("Yes" if Answer else "No")
